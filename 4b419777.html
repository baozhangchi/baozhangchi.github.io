C# 获取CPU、内存、硬盘IO和硬盘使用空间情况信息

Zhangchi Bao
03-16
开发 › C# › Windows
C#, Windows, 开发

你可以使用PerformanceCounter类来获取计算机系统的CPU使用率、内存使用率、磁盘I/O使用率和磁盘空间使用情况信息。

以下是一个示例代码，展示如何使用PerformanceCounter类来获取这些信息：

using System;
using System.Diagnostics;

namespace PerformanceCounterExample
{
    class Program
    {
        static void Main(string[] args)
        {
            // 获取CPU使用率计数器
            var cpuCounter = new PerformanceCounter("Processor", "% Processor Time", "_Total");

            // 获取可用内存计数器
            var memoryCounter = new PerformanceCounter("Memory", "Available MBytes");

            // 获取磁盘读取操作/秒计数器
            var diskReadCounter = new PerformanceCounter("PhysicalDisk", "Disk Read Bytes/sec", "_Total");

            // 获取磁盘写入操作/秒计数器
            var diskWriteCounter = new PerformanceCounter("PhysicalDisk", "Disk Write Bytes/sec", "_Total");

            // 获取可用磁盘空间计数器
            var diskSpaceCounter = new PerformanceCounter("LogicalDisk", "Free Megabytes", "C:");

            while (true)
            {
                // 获取CPU使用率
                var cpuUsage = cpuCounter.NextValue();

                // 获取可用内存
                var memoryUsage = memoryCounter.NextValue();

                // 获取磁盘读取操作字节/秒
                var diskRead = diskReadCounter.NextValue();

                // 获取磁盘写入操作字节/秒
                var diskWrite = diskWriteCounter.NextValue();

                // 获取可用磁盘空间
                var diskSpace = diskSpaceCounter.NextValue();

                // 输出信息
                Console.WriteLine("CPU使用率: {0}%", cpuUsage);
                Console.WriteLine("可用内存: {0}MB", memoryUsage);
                Console.WriteLine("磁盘读取操作字节/秒: {0}", diskRead);
                Console.WriteLine("磁盘写入操作字节/秒: {0}", diskWrite);
                Console.WriteLine("可用磁盘空间: {0}MB", diskSpace);

                // 等待1秒
                System.Threading.Thread.Sleep(1000);
            }
        }
    }
}

请注意，以上示例代码只获取计算机系统的总体情况。如果你想获取某个特定进程的CPU使用率或内存使用率，你需要使用特定进程的PerformanceCounter。
